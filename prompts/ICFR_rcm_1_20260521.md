### 5.2 파싱 로직 핵심

```python
# api/rcm.py 의 upload-excel 엔드포인트 내부

def parse_rcm_excel(file_bytes: bytes) -> ParsedRCM:
    """사이냅소프트 RCM 양식 파싱.
    
    가정:
    - 헤더는 7행에 걸쳐 있음 (1~7행)
    - 데이터는 8행부터 시작
    - B 컬럼 (프로세스번호) 가 빈 행은 데이터 끝
    """
    wb = load_workbook(BytesIO(file_bytes), read_only=True, data_only=True)
    ws = wb["RCM"]
    
    processes = {}      # code -> {name}
    sub_processes = {}  # code -> {name, process_code}
    risks = {}          # code -> {description, level, sub_process_code, owner}
    controls = []
    errors = []
    
    for row_idx, row in enumerate(ws.iter_rows(min_row=8, values_only=True), start=8):
        if row[1] is None:  # B 컬럼 (프로세스번호) 없으면 종료
            break
        
        try:
            # 그룹 1: 식별자 추출
            p_code = str(row[1]).strip()    # B
            p_name = str(row[2]).strip()    # C
            sp_code = str(row[3]).strip()   # D
            sp_name = str(row[4]).strip()   # E
            r_code = str(row[5]).strip()    # F
            c_code = str(row[6]).strip()    # G
            
            # 누적 (중복 자동 제거)
            processes[p_code] = p_name
            sub_processes[sp_code] = {"name": sp_name, "process_code": p_code}
            
            # 그룹 2: 위험·담당자
            risks[r_code] = {
                "description": str(row[8] or "").strip(),  # I
                "level": str(row[14] or "LR").strip(),     # O
                "sub_process_code": sp_code,
            }
            
            # 그룹 3+4+5: 통제
            control = {
                "code": c_code,
                "name": str(row[16] or "").strip(),         # Q
                "description": str(row[17] or "").strip(),  # R
                "objective": str(row[15] or "").strip(),    # P
                "owner_name": str(row[7] or "").strip(),    # H
                "risk_code": r_code,
                "is_key_control": (row[18] == "Yes"),       # S
                "preventive_detective": str(row[25] or "P").strip(),  # Z
                "auto_manual": str(row[26] or "M").strip(),           # AA
                # 활동 유형 6종 (T~Y)
                "activity_approval": (row[19] == "O"),
                "activity_verification": (row[20] == "O"),
                "activity_physical": (row[21] == "O"),
                "activity_master_data": (row[22] == "O"),
                "activity_reconciliation": (row[23] == "O"),
                "activity_supervision": (row[24] == "O"),
                # 7 assertions (AB~AH)
                "assertions": [
                    code for code, idx in [
                        ("E", 27), ("C", 28), ("R", 29), ("V", 30),
                        ("P", 31), ("O", 32), ("M", 33)
                    ] if row[idx] == "O"
                ],
                # 그룹 5
                "related_accounts": str(row[34] or "").strip(),  # AI
                "frequency": str(row[35] or "A").strip(),         # AJ
                "ipe_relevant": str(row[36] or "N/A").strip(),    # AK
                "related_systems": str(row[37] or "").strip(),    # AL
                "euc_description": str(row[38] or "").strip(),    # AM (N/A 도 그대로)
            }
            
            # 검증
            if not c_code:
                errors.append(f"Row {row_idx}: 통제활동번호 누락")
            if not control["name"]:
                errors.append(f"Row {row_idx}: 통제활동이름 누락")
            # ... 기타 검증 ...
            
            controls.append(control)
        except Exception as e:
            errors.append(f"Row {row_idx}: {str(e)}")
    
    wb.close()
    return ParsedRCM(
        processes=processes,
        sub_processes=sub_processes,
        risks=risks,
        controls=controls,
        errors=errors,
    )
```

### 5.3 검증 룰

### 5.4 멱등성·재실행 가능성

- **mode=preview**: 파싱만, DB 변경 없음
- **mode=commit**: 트랜잭션 안에서 INSERT
- **재실행**: 동일 코드 있으면 `IntegrityError` → 사용자에게 "기존 데이터 삭제 후 재업로드" 안내
- **부분 업로드 방지**: 트랜잭션 실패 시 전체 롤백

### 5.5 의존성 추가


### 6.2 구현 지침 (사용자 원칙: 코드 가볍게)

```python
# api/rcm.py

@router.get("/controls/search")
def search_controls(
    q: str | None = None,
    process_code: str | None = None,
    # ... 모든 필터 ...
    skip: int = 0,
    limit: int = 100,
    sort_by: str = "code",
    sort_order: str = "asc",
    db: Session = Depends(get_db),
    user: User = Depends(get_current_user),
):
    """단일 함수로 처리. SearchService 같은 추상화 금지."""
    query = db.query(Control)
    
    # 텍스트 검색
    if q:
        query = query.filter(
            or_(
                Control.code.ilike(f"%{q}%"),
                Control.name.ilike(f"%{q}%"),
                Control.description.ilike(f"%{q}%"),
            )
        )
    
    # 필터들 (None 이면 무시)
    if process_code:
        query = query.join(Risk).join(SubProcess).join(Process).filter(
            Process.code == process_code
        )
    # ... 기타 필터 ...
    
    # 정렬
    sort_col = getattr(Control, sort_by, Control.code)
    if sort_order == "desc":
        sort_col = sort_col.desc()
    query = query.order_by(sort_col)
    
    # 페이지네이션
    total = query.count()
    items = query.offset(skip).limit(limit).all()
    
    return {
        "items": items,
        "total": total,
        "skip": skip,
        "limit": limit,
        "sort": f"{sort_by}:{sort_order}",
    }
```

> ⚠️ **절대 만들지 말 것**:
> - ❌ `SearchService` 클래스
> - ❌ `QueryBuilder` 패턴
> - ❌ Generic search 라이브러리 도입
> - ❌ 동적 필터 추상화

→ **함수 1개 + 명시적 if 분기**. 끝.
